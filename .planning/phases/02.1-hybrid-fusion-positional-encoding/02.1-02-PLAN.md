**Section 4: Training Loop**
- Markdown cell: "## Training HybridGraphSAGE" with note about weight_decay for regularization on small dataset.
- Code cell: Define train and evaluate functions:
  ```python
  import torch.nn.functional as F

  def train_hybrid(model, loader, optimizer, device):
      model.train()
      total_loss = 0
      correct = 0
      total = 0
      for batch in loader:
          batch = batch.to(device)
          optimizer.zero_grad()
          logits = model(batch.x, batch.edge_index, batch.batch, batch.global_feat)
          loss = F.cross_entropy(logits, batch.y)
          loss.backward()
          optimizer.step()
          total_loss += loss.item() * batch.y.size(0)
          correct += (logits.argmax(dim=1) == batch.y).sum().item()
          total += batch.y.size(0)
      return total_loss / total, correct / total

  @torch.no_grad()
  def evaluate_hybrid(model, loader, device):
      model.eval()
      total_loss = 0
      correct = 0
      total = 0
      all_preds = []
      all_labels = []
      for batch in loader:
          batch = batch.to(device)
          logits = model(batch.x, batch.edge_index, batch.batch, batch.global_feat)
          loss = F.cross_entropy(logits, batch.y)
          total_loss += loss.item() * batch.y.size(0)
          preds = logits.argmax(dim=1)
          correct += (preds == batch.y).sum().item()
          total += batch.y.size(0)
          all_preds.extend(preds.cpu().tolist())
          all_labels.extend(batch.y.cpu().tolist())
      return total_loss / total, correct / total, all_preds, all_labels
  ```
- Code cell: Training loop with early stopping:
  ```python
  optimizer = torch.optim.Adam(hybrid_model.parameters(), lr=1e-3, weight_decay=1e-4)
  num_epochs = 50
  best_val_acc = 0
  patience = 10
  patience_counter = 0

  train_losses, val_losses = [], []
  train_accs, val_accs = [], []

  for epoch in range(num_epochs):
      train_loss, train_acc = train_hybrid(hybrid_model, hybrid_train_loader, optimizer, config.device)
      val_loss, val_acc, _, _ = evaluate_hybrid(hybrid_model, hybrid_val_loader, config.device)

      train_losses.append(train_loss)
      val_losses.append(val_loss)
      train_accs.append(train_acc)
      val_accs.append(val_acc)

      if val_acc > best_val_acc:
          best_val_acc = val_acc
          patience_counter = 0
          torch.save(hybrid_model.state_dict(), "hybrid_graphsage_best.pt")
      else:
          patience_counter += 1

      if (epoch + 1) % 5 == 0:
          print(f"Epoch {epoch+1}/{num_epochs} | "
                f"Train Loss: {train_loss:.4f} Acc: {train_acc:.4f} | "
                f"Val Loss: {val_loss:.4f} Acc: {val_acc:.4f}")

      if patience_counter >= patience:
          print(f"Early stopping at epoch {epoch+1}")
          break

  print(f"\nBest validation accuracy: {best_val_acc:.4f}")
  ```
